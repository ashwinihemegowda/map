# A Complete Graph
x = np.linspace(0, 2 * np.pi, 100)
y = np.sin(x)
z = np.cos(x)

# Title
plt.title("SIN and Cos Graph")

# Sin Curve Properties
sin_props = {
    'color': 'r',
    'linestyle': 'dashed',
    'linewidth': 2.0,
    'marker': 'o',
    'markeredgecolor': 'blue',
    'markerfacecolor': 'cyan',
    'markevery': 10,
    'label': 'SIN(X)'
}

# Cos Curve Properties
cos_props = {
    'color': 'g',
    'linestyle': 'dashdot',
    'linewidth': 2.0,
    'marker': '^',
    'markeredgecolor': 'red',
    'markerfacecolor': 'yellow',
    'markevery': 5,
    'label': 'COS(X)'
}

# Plot Sin Curve
plt.plot(x,
         y,
         **sin_props)

# Plot Cos Curve
plt.plot(x,
         z,
         **cos_props)

# Legend
plt.legend(title='Sin and Cos Curves', ncol=1)

# Labels for X-Axis and Y-Axis
plt.xlabel('Random Values')
plt.ylabel('Sin and Cos Values')

# Save the plot
plt.savefig('Sin_and_Cos_Curves.png')
