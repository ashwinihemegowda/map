#Finding Z Score => (data_point-mean)/standard_deviation
mean_col_1=np.mean(df2['col_1'])
print('mean_col_1 -',mean_col_1)
std_col_1 = np.std(df2['col_1'])
print('std_col_1 -',std_col_1)
z_score_col_1 = np.abs(stats.zscore(df2['col_1']))
print('Outliers for col_1 -',np.where(z_score_col_1 >3))
print('-'*60)

#Finding IQR =>Quartile3 - Quartile1
Q1_col_1 = np.percentile(df2['col_1'],25,method='midpoint')
Q3_col_1 = np.percentile(df2['col_1'],75,method='midpoint')
IQR_col_1 = Q3_col_1 - Q1_col_1
print('IQR_col_1 -',IQR_col_1)
#Upper and Lower bound
upper_col_1 = Q3_col_1 + 1.5 * IQR_col_1
lower_col_1 = Q1_col_1 - 1.5 * IQR_col_1
